function renderFeatured() {
  const featured = TOOLS.filter(t => t.featured);
  const grid = document.getElementById('featured-grid');
  grid.innerHTML = featured.map(tool => `
    <div class="tool-card" data-id="${tool.id}">
      <div class="card-top">
        <span class="cat-badge ${tool.category}">${tool.category}</span>
        <span class="price-badge ${tool.pricing}">${tool.pricing}</span>
      </div>
      <div class="card-body">
        <a href="tool.html?id=${tool.id}">
          <p class="card-name">${tool.name}</p>
        </a>
        <p class="card-desc">${tool.description}</p>
      </div>
      <div class="card-actions">
        <button class="btn btn-save" data-id="${tool.id}">Save</button>
        <a href="${tool.url}" target="_blank" class="btn">Visit</a>
      </div>
    </div>
  `).join('');
}

11 Mauricio Add save / unsave with localStorage ›

Add one click listener to the #featured-grid container (event delegation). When a save button is clicked, read the tool ID from data-id, read the saved array from localStorage, toggle the ID in or out, write it back, and update the button state.

Code hint — localStorage toggle 
function getSaved() {
  return JSON.parse(localStorage.getItem('savedTools') || '[]');
}

document.getElementById('featured-grid')
  .addEventListener('click', e => {
    if (!e.target.matches('.btn-save')) return;
    const id = parseInt(e.target.dataset.id);
    let saved = getSaved();
    saved = saved.includes(id)
      ? saved.filter(i => i !== id)
      : [...saved, id];
    localStorage.setItem('savedTools', JSON.stringify(saved));
    e.target.textContent = saved.includes(id) ? 'Saved' : 'Save';
  });

Cross-page state — data Mauricio's home page saves into localStorage can be read here on a completely different page. This is how web apps share state without a backend server.
Code hint — read saved tools 
function loadSaved() {
  const ids = JSON.parse(localStorage.getItem('savedTools') || '[]');

  if (ids.length === 0) {
    document.getElementById('empty-state').style.display = 'block';
    return;
  }

  const tools = ids.map(id => getToolById(id)).filter(Boolean);
  document.getElementById('saved-count').textContent = tools.length;
  document.getElementById('saved-grid').innerHTML = tools.map(tool =>
    /* same card HTML as home page */
    `<div class="tool-card">...</div>`
  ).join('');
}
